CREATE PROC [dbo].[spIsGetCompanySymbolExist]
		@SYMBOL varchar(500)
		AS
BEGIN
	DECLARE @Count bigint;
	SELECT @Count = COUNT(*) FROM [SCRIP_INFO] WHERE [SI_SYMBOL] = @SYMBOL 
	SELECT @Count AS [FUND_EXIST]
END
