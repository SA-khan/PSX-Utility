CREATE OR REPLACE PROCEDURE 
spINSERT_MARKET_SUMMARY_CLOSE
(
  ID IN number,
  DATE_ IN DATE,
  SYMBOL IN varchar2,
  CATEGORY IN number,
  NAME IN varchar2,
  OPEN IN varchar2,
  HIGH IN varchar2,
  LOW IN varchar2,
  CLOSING IN varchar2,
  VOLUME IN varchar2,
  LDCP IN varchar2
  )
AS
BEGIN
  INSERT INTO MARKET_SUMMARY_CLOSING (MSC_ID,MSC_DATE,MSC_SYMBOL,MSC_CATEGORY,MSC_NAME,MSC_OPEN,MSC_HIGH,MSC_LOW,MSC_CLOSING,MSC_VOLUME,MSC_LDCP) VALUES (ID,DATE_,SYMBOL,CATEGORY,NAME,OPEN,HIGH,LOW,CLOSING,VOLUME,LDCP);
END;
