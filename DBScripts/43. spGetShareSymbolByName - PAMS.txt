CREATE PROC [dbo].[spGetShareSymbolByName]
		@NAME varchar(500)
AS
BEGIN
SELECT 
      [share_symbol] AS [C_SYMBOL]
  FROM [pms_shares] WHERE [share_name] LIKE '%'+@NAME+'%'
END
