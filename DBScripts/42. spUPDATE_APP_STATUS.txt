CREATE PROC [dbo].[spUPDATE_APP_STATUS]
	@ID int,
	@APS_TASK_DATE datetime,
	@APS_TABLE varchar(500),
	@APS_STATUS varchar(MAX),
	@APS_CLEARING_TABLE bit,
	@APS_FETCHING_DATA bit,
	@APS_DUMPING_DATA bit,
	@APS_FETCH_DATA_START_TIME datetime,
	@APS_FETCH_DATA_END_TIME datetime,
	@APS_DUMP_DATA_TIME datetime,
	@APS_COMMENTS varchar(MAX)
AS
BEGIN

UPDATE [dbo].[APP_STATUS]
   SET [APS_TASK_DATE] = @APS_TASK_DATE
      ,[APS_TABLE] = @APS_TABLE
      ,[APS_STATUS] = @APS_STATUS
      ,[APS_CLEARING_TABLE] = @APS_CLEARING_TABLE
      ,[APS_FETCHING_DATA] = @APS_FETCHING_DATA
      ,[APS_DUMPING_DATA] = @APS_DUMPING_DATA
      ,[APS_FETCH_DATA_START_TIME] = @APS_FETCH_DATA_START_TIME
      ,[APS_FETCH_DATA_END_TIME] = @APS_FETCH_DATA_END_TIME
      ,[APS_DUMP_DATA_TIME] = @APS_DUMP_DATA_TIME
      ,[APS_COMMENTS] = @APS_COMMENTS
 WHERE APS_ID = 1

END
