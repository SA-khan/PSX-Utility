CREATE PROC spUPDATE_FUND_SCRIP_BUCKET
	@FSB_DATE Datetime,
	@FSB_STATUS varchar(max),
	@FSB_FUND_ID bigint,
	@FSB_FUND_NAME varchar(max),
	@FSB_SHARE_NAME varchar(max),
	@FSB_SHARE_SYMBOL varchar(500),
	@FSB_SHARE_QUANTITY decimal(18,2),
	@FSB_SHARE_AVG_PRICE decimal(18,2),
	@FSB_SHARE_BOOK_COST decimal(18,2),
	@FSB_SHARE_MARKET_PRICE decimal(18,2),
	@FSB_SHARE_MARKET_VALUE decimal(18,2),
	@FSB_SHARE_APP_DEP decimal(18,2),
	@FSB_SHARE_PERCENTAGE_CLOSING decimal(18,2)
AS
BEGIN
UPDATE [dbo].[FUND_SCRIP_BUCKET]
   SET [FSB_DATE] = @FSB_DATE
      ,[FSB_STATUS] = @FSB_STATUS
      ,[FSB_FUND_ID] = @FSB_FUND_ID
      ,[FSB_FUND_NAME] = @FSB_FUND_NAME
      ,[FSB_SHARE_NAME] = @FSB_SHARE_NAME
      ,[FSB_SHARE_SYMBOL] = @FSB_SHARE_SYMBOL
      ,[FSB_SHARE_QUANTITY] = @FSB_SHARE_QUANTITY
      ,[FSB_SHARE_AVG_PRICE] = @FSB_SHARE_AVG_PRICE
      ,[FSB_SHARE_BOOK_COST] = @FSB_SHARE_BOOK_COST
      ,[FSB_SHARE_MARKET_PRICE] = @FSB_SHARE_MARKET_PRICE
      ,[FSB_SHARE_MARKET_VALUE] = @FSB_SHARE_MARKET_VALUE
      ,[FSB_SHARE_APP_DEP] = @FSB_SHARE_APP_DEP
      ,[FSB_SHARE_PERCENTAGE_CLOSING] = @FSB_SHARE_PERCENTAGE_CLOSING
 WHERE [FSB_SHARE_SYMBOL] = @FSB_SHARE_SYMBOL
END
