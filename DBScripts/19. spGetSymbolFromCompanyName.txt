ALTER PROC [dbo].[spGetSymbolFromCompanyName]
		@CompanyName varchar(500)
		AS
		BEGIN
		DECLARE @value varchar(500)
			SELECT @value = C_SYMBOL FROM COMPANY WHERE C_NAME = @CompanyName
			if @value is null
			 SELECT 'Nil' AS C_SYMBOL 
			else
			   SELECT @value AS C_SYMBOL
		END
