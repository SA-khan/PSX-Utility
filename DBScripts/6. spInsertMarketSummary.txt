ALTER PROC [dbo].[spInsertMarketSummary] 
		@COMPANY_NAME varchar(MAX),
		@COMPANY_SYMBOL varchar(MAX),
		@COMPANY_LDCP float,
		@COMPANY_OPEN float,
		@COMPANY_HIGH float,
		@COMPANY_LOW float,
		@COMPANY_CURRENT float,
		@COMPANY_CHANGE float,
		@COMPANY_VOLUME float
	AS
BEGIN

INSERT INTO [dbo].[CURRENT_MARKET_SUMMARY]
           ([CMS_SECTOR]
		   ,[CMS_SCRIP_NAME]
           ,[CMS_SYMBOL]
           ,[CMS_LDCP]
           ,[CMS_OPEN]
           ,[CMS_HIGH]
           ,[CMS_LOW]
           ,[CMS_CURRENT]
           ,[CMS_CHANGE]
           ,[CMS_VOLUME])
     VALUES
           (''
		   ,UPPER(@COMPANY_NAME)
           ,UPPER(@COMPANY_SYMBOL)
           ,@COMPANY_LDCP
           ,@COMPANY_OPEN
           ,@COMPANY_HIGH 
           ,@COMPANY_LOW
           ,@COMPANY_CURRENT
           ,@COMPANY_CHANGE
           ,@COMPANY_VOLUME)
END
